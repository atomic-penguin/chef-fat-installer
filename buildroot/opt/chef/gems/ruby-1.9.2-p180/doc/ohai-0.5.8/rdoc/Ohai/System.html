181:       unless force
182:         return true if @seen_plugins[plugin_name]
183:       end
184: 
185:       if Ohai::Config[:disabled_plugins].include?(plugin_name)

187:         return false
188:       end

191:

193: 
194:       Ohai::Config[:plugin_path].each do |path|
195:         check_path = File.expand_path(File.join(path, filename))
196:         begin
197:           @seen_plugins[plugin_name] = true
198:           Ohai::Log.debug("Loading plugin #{plugin_name}")
199:           from_file(check_path)
200:           return true
201:         rescue IOError => e
202:           Ohai::Log.debug("No #{plugin_name} at #{check_path}")
203:         rescue Exception,Errno::ENOENT => e
204:           Ohai::Log.debug("Plugin #{plugin_name} threw exception #{e.inspect} #{e.backtrace.join("\n")}")
