helper to define abstract method in Ruby.

example1. (shorter notation) 

  require 'abstract'
  class Foo
    abstract_method 'arg1, arg2=""', :method1, :method2, :method3
  end

example2. (RDoc friendly notation) 

  require 'abstract'
  class Bar
    # ... method1 description ...
    def method1(arg1, arg2="")
      not_implemented
    end

    # ... method2 description ...
    def method2(arg1, arg2="")
      not_implemented
    end
  end
